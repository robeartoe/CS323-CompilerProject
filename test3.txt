  real principal$,maxPrincipal,salesTax; !REAL Declarations!
  int step; !INT Declarations!

  get(principal$,maxPrincipal,salesTax,step);
  while(step < maxPrincipal)
    {
      put(principal);
      put(salesTax);
      put(principal + salesTax);
      step = step + 1;
    }

    if (step == maxPrincipal)
      put(true);
    else
      put(false);
    endif
